# if ubuntu
sudo add-apt-repository ppa:alex-p/tesseract-ocr
# if debian
sudo apt-get install -y dirmngr build-essential libfontconfig1 libicu55 liblept5
sudo vi /etc/apt/sources.list.d/alex-p-tesseract-ocr.list
